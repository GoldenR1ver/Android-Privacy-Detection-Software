\section{实验背景}
\begin{frame}
    \frametitle{实验背景}
    \begin{itemize}
        \item \textbf{实验任务}
        \begin{itemize}
            \item 对比 Android 应用 \texttt{Data Safety} 与 \texttt{Privacy Policy}
            \item 利用 LLM 自动识别合规风险标签
        \end{itemize}
        \item \textbf{输出标签（三分类）}
        \begin{itemize}
            \item \texttt{incorrect}：Data Safety 未声明，但隐私政策提到
            \item \texttt{incomplete}：Data Safety 有声明，但细节不完整
            \item \texttt{inconsistent}：两者都提到，但描述冲突
        \end{itemize}
        \item \textbf{实验价值}
        \begin{itemize}
            \item 构建“模型初筛 + 人工复核”的隐私审计流程
            \item 将输出结果映射到你们既有22项合规框架
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]
    \frametitle{Prompt 怎么写（代码中的真实模板）}
    \scriptsize
\begin{lstlisting}[language=python]
SYSTEM_PROMPT = "You are an expert in analyzing..."

USER_PROMPT_TEMPLATE = """
Classify three labels:
1) incorrect ...
2) incomplete ...
3) inconsistent ...

Output format strictly as JSON:
{"incorrect": 0 or 1, "incomplete": 0 or 1, "inconsistent": 0 or 1}
Return JSON only.

Data Safety:
{data_safety}
Privacy Policy:
{privacy_policy}
"""
\end{lstlisting}
    \vspace{-0.4em}
    \begin{itemize}
        \item \textbf{设计思路}：角色约束（system）+ 任务约束（user）+ 格式约束（JSON only）
        \item \textbf{工程设置}：\texttt{temperature=0}，降低随机性，提高可复现性
    \end{itemize}
\end{frame}

        误判原因：Data Safety与隐私政策的冲突表达较隐式，模型更偏向保守预测0。
    \end{block}
    \begin{block}{案例2：正确样本}
        样本ID：1（Text On Photo Photo Per Status）；预测与真值均为(incorrect=1, incomplete=1, inconsistent=0)。\\
        关键证据：Data Safety几乎为空，但隐私政策明确提到数据共享与数据收集。
    \end{block}
    \begin{block}{小结}
        模型对“显式缺失声明”识别较稳定；对“文本冲突”类型仍需人工复核。
    \end{block}

\section{合规解释与讨论}
\begin{frame}
    \frametitle{标签到22项标准的映射}
    \begin{itemize}
        \item \textbf{\texttt{incorrect}}：优先关联第2、9、13、20项（缺失声明风险）
        \item \textbf{\texttt{incomplete}}：保存期限、接收方信息、用户权利路径粒度不足
        \item \textbf{\texttt{inconsistent}}：重点核查跨境传输与第三方共享冲突
        \item 本实验结果与Week2结论部分一致：声明缺失更易检出，复杂冲突仍是主要难点
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{关键发现}
    \begin{itemize}
        \item \textbf{发现1}：\texttt{incorrect}/\texttt{incomplete} 均为高Precision、低Recall，说明“命中较准但漏检明显”
        \item \textbf{发现2}：\texttt{inconsistent} 的 P/R/F1 为0，表明当前模型对冲突语义识别能力不足
        \item \textbf{发现3}：需结合混淆矩阵解读，不能仅依据Accuracy判断模型可用性
        \item \textbf{发现4}：现阶段流程适合做合规初筛，不适合直接替代人工审计
    \end{itemize}
\end{frame}

\section{结论与改进}
\begin{frame}
    \frametitle{实验结论}
    \begin{itemize}
        \item 结论1：实验流程在150条样本上可复现并稳定运行，输出结构完整可评估
        \item 结论2：当前表现最佳标签为\texttt{incomplete}（F1=0.523），最难标签为\texttt{inconsistent}（F1=0）
        \item 结论3：与Week2的22项研究结论部分一致：显式缺失较易识别，复杂合规冲突更难自动检测
        \item 结论4：推荐采用“LLM初筛 + 人工复核”的工程化方案
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{改进方向}
    \begin{itemize}
        \item 模型：继续对比 Qwen/DeepSeek，并重点优化\texttt{inconsistent}标签
        \item Prompt：加入冲突判定few-shot样例，明确“矛盾表达”判断规则
        \item 数据：扩充冲突类样本并平衡标签分布，降低模型偏向负类的风险
        \item 工程：增加异常重试、JSON校验与复核队列，形成稳定审计流水线
    \end{itemize}
\end{frame}
